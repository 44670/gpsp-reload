cmake_minimum_required(VERSION 3.22)

set(USE_QEMU "0" CACHE BOOL "Build for ESP-IDF QEMU instead of CoreS3 SE hardware")
set(USE_DEBUG "0" CACHE BOOL "Enable ESP32-S3 debugger and instruction trace hooks")
set(USE_DYNAREC "1" CACHE BOOL "Enable ESP32-S3 native dynarec")
set(GPSP_TEST_FRAMES "300" CACHE STRING "Number of frames to run in frames mode")
set(GPSP_TEST_EXPECT_FB_HASH "0" CACHE STRING "Expected FNV-1a framebuffer stream hash, or 0 to skip")
set(GPSP_TEST_DUMP_FRAME "0" CACHE STRING "Emit the last RGB565 frame over USB Serial/JTAG for host PNG capture")
set(GPSP_PC_SAMPLER "0" CACHE BOOL "Enable random ESP32-S3 raw host PC sampler")
set(GPSP_PC_SAMPLER_MIN_MS "50" CACHE STRING "Minimum random PC sampler delay in milliseconds")
set(GPSP_PC_SAMPLER_MAX_MS "250" CACHE STRING "Maximum random PC sampler delay in milliseconds")
set(GPSP_PC_SAMPLER_REPORT_MS "5000" CACHE STRING "PC sampler report period in milliseconds")
set(GPSP_PC_SAMPLER_VERBOSE "0" CACHE BOOL "Print every raw PC sampler record instead of only histogram reports")
set(GPSP_DISABLE_INTERP "0" CACHE BOOL "Disable helper-backed/interpreter fallback blocks in ESP32-S3 dynarec tests")
if(USE_QEMU)
    if(NOT SDKCONFIG)
        set(SDKCONFIG "${CMAKE_BINARY_DIR}/sdkconfig")
    endif()
    if(NOT SDKCONFIG_DEFAULTS)
        set(SDKCONFIG_DEFAULTS
            "${CMAKE_CURRENT_LIST_DIR}/sdkconfig.defaults"
            "${CMAKE_CURRENT_LIST_DIR}/sdkconfig.qemu.defaults")
    endif()
endif()
if(NOT DEFINED GPSP_CORES3SE_LCD OR GPSP_CORES3SE_LCD STREQUAL "")
    if(USE_QEMU)
        set(GPSP_CORES3SE_LCD "0" CACHE BOOL "Enable CoreS3 SE hardware init and LCD output")
    else()
        set(GPSP_CORES3SE_LCD "1" CACHE BOOL "Enable CoreS3 SE hardware init and LCD output")
    endif()
endif()

include($ENV{IDF_PATH}/tools/cmake/project.cmake)
idf_build_set_property(MINIMAL_BUILD ON)
project(gpsp_esp32s3)
